
//3516. Find Closest Person
method-direct logic 
int findClosest(int x, int y, int z) {
        int dist1=abs(z-x);///for 1st to 3rd person
        int dist2=abs(z-y);//for 2nd to 3rd person
        if(dist1<dist2){
            return 1;
        }else if(dist1>dist2){
            return 2;
        }
        return 0;
    }
