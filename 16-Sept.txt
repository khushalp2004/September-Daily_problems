

///2197. Replace Non-Coprime Numbers in Array

vector<int> replaceNonCoprimes(vector<int>& nums) {
        ///left to right
        ///vector as a stack

        vector<int> result;

        for(int num: nums){
            while(!result.empty()){
                int prev=result.back();
                int curr=num;

                int GCD=gcd(prev,curr);////get GCD
                if(GCD==1){
                    break;
                }
                result.pop_back();

                int LCM=prev/GCD*curr;
                num=LCM;///merged number
            }
            result.push_back(num);//merged number push back to result

        }
        return result;
    }
