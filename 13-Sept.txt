
///3541. Find Most Frequent Vowel and Consonant

bool isVowel(char c){
        return (c=='a' || c=='e' || c=='i' || c=='o' || c=='u');
    }
    int maxFreqSum(string s) {
        unordered_map<char,int> vowMap,conMap;
        //push vowels in map
        for(char c: s){
            if(isVowel(c)){
                vowMap[c]++;
            }
        }

        //push Consonants in map
        for(char c: s){
            if(!isVowel(c)){
                conMap[c]++;
            }
        }
        
        //find maximum occuring vowel count
        int maxFreqVow=0;
        for(auto &p: vowMap){
            maxFreqVow=max(maxFreqVow,p.second);
        }

        //find maximum occuring consonant count
        int maxFreqCons=0;
        for(auto &p: conMap){
            maxFreqCons=max(maxFreqCons,p.second);
        }

        ///return Max(Vowel) + Max(Consonant)
        return maxFreqVow+maxFreqCons;
    }
