

////1317. Convert Integer to the Sum of Two No-Zero Integers
///first check non zero in integer then only push_back it

bool isNonZero(int num){
        while(num>0){
            if(num%10==0){
                return false;
            }
            num/=10;
        }
        return true;
    }
    vector<int> getNoZeroIntegers(int n) {
        vector<int> result;
        for(int a=1;a<n;a++){
            int b=n-a;
            if(isNonZero(a) && isNonZero(b)){//check here the non zero integers
                result.push_back(a);
                result.push_back(b);
                break;
            }
        }
        return result;
    }
