
//611. Valid Triangle Number

int triangleNumber(vector<int>& nums) {
        int count=0;
        sort(nums.begin(),nums.end());
        for(int k=nums.size()-1;k>1;k--){
            int start=0, end=k-1;
            while(start<end){
                if(nums[start]+nums[end]>nums[k]){
                    count+=end-start;
                    end--;
                }else{
                    start++;
                }
            }
        }
        return count;
    }
