
///3005. Count Elements With Maximum Frequency
//using unordered_map store their Frequencies 
int maxFrequencyElements(vector<int>& nums) {
        unordered_map<int,int> mp;
        for(int num: nums){
            mp[num]++;
        }
        int maxFreq=0;
        for(auto &it: mp){
            maxFreq=max(maxFreq,it.second);
        }

        int result=0;
        for(auto &it: mp){
            if(it.second==maxFreq){
                result+=it.second;
            }
        }

        return result;
    }
